```sql
-- vw_cjm_pipeline_flow: aggregados por stage (para o mapa visual)
CREATE OR REPLACE VIEW vw_cjm_pipeline_flow AS
SELECT
  js.location_id,
  js.pipeline_id,
  sc.pipeline_name,
  js.current_stage,
  sc.stage_name,
  sc.stage_order,
  sc.color,
  sc.owner_name,
  sc.sla_hours,
  COUNT(js.contact_id)                                    AS contact_count,
  COALESCE(
    AVG(business_hours_diff(js.entered_stage_at, now())),
    0
  )                                                       AS avg_hours_in_stage,
  COUNT(*) FILTER (WHERE js.sla_status = 'breach')        AS sla_breach_count,
  COUNT(*) FILTER (WHERE js.sla_status = 'warning')       AS sla_warning_count
FROM cjm_journey_state js
LEFT JOIN cjm_stage_config sc
  ON sc.location_id  = js.location_id
  AND sc.pipeline_id  = js.pipeline_id
  AND sc.stage_id     = js.current_stage
GROUP BY
  js.location_id, js.pipeline_id, sc.pipeline_name,
  js.current_stage, sc.stage_name, sc.stage_order,
  sc.color, sc.owner_name, sc.sla_hours;

-- vw_cjm_sla_status: contacts em warning/breach ordenados por horas atrasadas
CREATE OR REPLACE VIEW vw_cjm_sla_status AS
SELECT
  js.location_id,
  js.contact_id,
  js.pipeline_id,
  sc.pipeline_name,
  js.current_stage,
  sc.stage_name,
  sc.sla_hours,
  js.sla_status,
  js.entered_stage_at,
  business_hours_diff(js.entered_stage_at, now()) AS hours_in_stage,
  GREATEST(
    business_hours_diff(js.entered_stage_at, now()) - COALESCE(sc.sla_hours, 999),
    0
  )                                               AS hours_overdue
FROM cjm_journey_state js
LEFT JOIN cjm_stage_config sc
  ON sc.location_id  = js.location_id
  AND sc.pipeline_id  = js.pipeline_id
  AND sc.stage_id     = js.current_stage
WHERE js.sla_status IN ('warning', 'breach')
ORDER BY hours_overdue DESC;
```

**business_hours_diff() — implementação completa (do RESEARCH.md):**

```sql
CREATE OR REPLACE FUNCTION business_hours_diff(
  ts_start   timestamptz,
  ts_end     timestamptz,
  tz         text    DEFAULT 'America/Sao_Paulo',
  work_start time    DEFAULT '09:00',
  work_end   time    DEFAULT '18:00'
) RETURNS numeric AS $$
DECLARE
  v_hours          numeric := 0;
  v_day            date;
  v_day_start      timestamptz;
  v_day_end        timestamptz;
  v_overlap_start  timestamptz;
  v_overlap_end    timestamptz;
BEGIN
  IF ts_start > ts_end THEN RETURN 0; END IF;
  FOR v_day IN
    SELECT generate_series(
      (ts_start AT TIME ZONE tz)::date,
      (ts_end   AT TIME ZONE tz)::date,
      '1 day'::interval
    )::date
  LOOP
    IF EXTRACT(DOW FROM v_day) IN (0, 6) THEN CONTINUE; END IF;
    IF EXISTS (SELECT 1 FROM business_calendar WHERE calendar_date = v_day AND is_holiday = true)
    THEN CONTINUE; END IF;
    v_day_start     := (v_day + work_start) AT TIME ZONE tz;
    v_day_end       := (v_day + work_end)   AT TIME ZONE tz;
    v_overlap_start := GREATEST(ts_start, v_day_start);
    v_overlap_end   := LEAST(ts_end, v_day_end);
    IF v_overlap_end > v_overlap_start THEN
      v_hours := v_hours + EXTRACT(EPOCH FROM (v_overlap_end - v_overlap_start)) / 3600.0;
    END IF;
  END LOOP;
  RETURN v_hours;
END;
$$ LANGUAGE plpgsql STABLE;
```
